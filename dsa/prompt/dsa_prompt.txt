func eraseOverlapIntervals(intervals [][]int) int {
	fmt.Println("ðŸ“Œ Input intervals:", intervals)

	if len(intervals) == 0 {

	}

	// Step 1: Sort intervals by end time
	sort.Slice(intervals, func(i, j int) bool {
		return intervals[i][1] < intervals[j][1]
	})
	fmt.Println("\nðŸ”¹ After sorting by end time:", intervals)

	// Step 2: Pick the first interval
	count := 0
	prevEnd := intervals[0][1]
	fmt.Printf("ðŸ‘‰ Start with interval %v, prevEnd = %d\n", intervals[0], prevEnd)

	// Step 3: Iterate and check overlaps
	for i := 1; i < len(intervals); i++ {
		curr := intervals[i]
		fmt.Printf("\nChecking interval %v: ", curr)

		if curr[0] < prevEnd {
			// Overlap â†’ remove this interval
			count++
			fmt.Printf("âŒ Overlaps with prevEnd=%d â†’ Remove (total removed=%d)", prevEnd, count)
		} else {
			// No overlap â†’ keep it
			prevEnd = curr[1]
			fmt.Printf("âœ… No overlap â†’ Keep, update prevEnd=%d", prevEnd)
		}
	}

	// Step 4: Result
	fmt.Printf("\n\nâœ… Final Answer: Need to remove %d interval(s)\n", count)
	return count
}
